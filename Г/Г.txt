<script>
  const pairsData = [
    { id: 1, word: 'гамак', image: '1.png' },
    { id: 2, word: 'гарбуз', image: '2.png' },
    { id: 3, word: 'голуб', image: '3.png' },
    { id: 4, word: 'гребінець', image: '4.png' },
    { id: 5, word: 'гриби', image: '5.png' },
    { id: 6, word: 'гуска', image: '6.png' }
  ];

  const colors = ['#4da6ff','#22c55e','#a855f7','#f59e0b','#06b6d4', '#ef4444'];

  const colLeft = document.getElementById('col-left');
  const colCenter = document.getElementById('col-center');
  const colRight = document.getElementById('col-right');
  const canvas = document.getElementById('line-canvas');
  const winMsg = document.getElementById('winMsg');

  let selectedWord = null;
  let matched = 0;

  function initGame() {
    colLeft.innerHTML = '';
    colCenter.innerHTML = '';
    colRight.innerHTML = '';

    const defs = canvas.querySelector('defs');
    canvas.innerHTML = '';
    if (defs) canvas.appendChild(defs);

    matched = 0;
    selectedWord = null;
    winMsg.style.display = 'none';

    // 1. Зображення (ліворуч і праворуч)
    const shuffledImages = [...pairsData].sort(() => Math.random() - 0.5);
    const half = Math.ceil(shuffledImages.length / 2);
    const leftImages = shuffledImages.slice(0, half);
    const rightImages = shuffledImages.slice(half);

    leftImages.forEach(p => createImgItem(p, colLeft));
    rightImages.forEach(p => createImgItem(p, colRight));

    // 2. Слова (центр)
    [...pairsData].sort(() => Math.random() - 0.5).forEach(p => {
      const d = document.createElement('div');
      d.className = 'item item-word';
      d.dataset.id = p.id;
      d.dataset.color = colors[p.id - 1];
      d.textContent = p.word;
      d.onclick = () => selectWord(d);
      colCenter.appendChild(d);
    });
  }

  function createImgItem(data, container) {
    const d = document.createElement('div');
    d.className = 'item item-image';
    d.dataset.id = data.id;
    d.innerHTML = `<img src="${data.image}" alt="${data.word}">`;
    d.onclick = () => selectImage(d);
    container.appendChild(d);
  }

  function selectWord(el) {
    if (el.classList.contains('matched')) return;
    document.querySelectorAll('.item-word').forEach(i => i.classList.remove('selected'));
    el.classList.add('selected');
    selectedWord = el;
  }

  function selectImage(el) {
    if (!selectedWord || el.classList.contains('matched')) return;

    if (el.dataset.id === selectedWord.dataset.id) {
      drawArrow(selectedWord, el, selectedWord.dataset.color);

      selectedWord.classList.remove('selected');
      selectedWord.classList.add('matched');
      el.classList.add('matched');

      selectedWord.onclick = null;
      el.onclick = null;

      selectedWord = null;
      matched++;

      document.getElementById('soundCorrect').play();

      if (matched === pairsData.length) {
        winMsg.style.display = 'flex';
        setTimeout(() => winMsg.style.display = 'none', 3000);
      }
    } else {
      el.classList.add('error');
      document.getElementById('soundWrong').play();
      setTimeout(() => el.classList.remove('error'), 400);
    }
  }

  function drawArrow(wordEl, imgEl, color) {
    const wRect = wordEl.getBoundingClientRect(); 
    const iRect = imgEl.getBoundingClientRect();  
    const wrap = document.querySelector('.game-wrapper').getBoundingClientRect();

    const l = document.createElementNS('http://www.w3.org/2000/svg', 'line');
    
    const y1 = wRect.top + wRect.height / 2 - wrap.top;
    const y2 = iRect.top + iRect.height / 2 - wrap.top;
    let x1, x2;

    if (iRect.left < wRect.left) {
      x1 = wRect.left - wrap.left;
      x2 = iRect.right - wrap.left;
    } else {
      x1 = wRect.right - wrap.left;
      x2 = iRect.left - wrap.left;
    }

    l.setAttribute('x1', x1);
    l.setAttribute('y1', y1);
    l.setAttribute('x2', x2);
    l.setAttribute('y2', y2);
    
    l.setAttribute('stroke', color);
    l.setAttribute('stroke-width', '4');
    l.setAttribute('marker-end', 'url(#arrowhead)');
    canvas.appendChild(l);
  }

  document.getElementById('btnReset').onclick = initGame;
  document.getElementById('btnSound').onclick = () => {
    const a = document.getElementById('audioTask');
    a.currentTime = 0;
    a.play();
  };

  initGame();
</script>
