  <script>
    const TASK_TEXT = "Із звуків склади назву музичного інструмента і познач відповідь справа біля малюнка.";

    const ROWS = [
      {
        key: "row1",
        word: ["с","к","р","и","п","к","а"],
        imageSrc: "i1.png",
        letterSounds: [
          { ch:"с", audio:"audio1_1" },
          { ch:"к", audio:"audio1_2" },
          { ch:"р", audio:"audio1_3" },
          { ch:"и", audio:"audio1_4" },
          { ch:"п", audio:"audio1_5" },
          { ch:"к", audio:"audio1_6" },
          { ch:"а", audio:"audio1_7" }
        ],
        slovoAudio: "slovo1"
      },
      {
        key: "row2",
        word: ["к","о","б","з","а"],
        imageSrc: "i2.png",
        letterSounds: [
          { ch:"к", audio:"audio2_1" },
          { ch:"о", audio:"audio2_2" },
          { ch:"б", audio:"audio2_3" },
          { ch:"з", audio:"audio2_4" },
          { ch:"а", audio:"audio2_5" }
        ],
        slovoAudio: "slovo2"
      },
      {
        key: "row3",
        word: ["б","а","н","д","у","р","а"],
        imageSrc: "i3.png",
        letterSounds: [
          { ch:"б", audio:"audio3_1" },
          { ch:"а", audio:"audio3_2" },
          { ch:"н", audio:"audio3_3" },
          { ch:"д", audio:"audio3_4" },
          { ch:"у", audio:"audio3_5" },
          { ch:"р", audio:"audio3_6" },
          { ch:"а", audio:"audio3_7" }
        ],
        slovoAudio: "slovo3"
      }
    ];

    const state = {};
    let dragTokenId = null;

    function shuffle(arr){
      const a = arr.slice();
      for(let i=a.length-1;i>0;i--){
        const j = Math.floor(Math.random()*(i+1));
        [a[i],a[j]]=[a[j],a[i]];
      }
      return a;
    }

    function playAudioById(id){
      const el = document.getElementById(id);
      if(!el) return;
      try{ el.currentTime = 0; el.play(); }catch(e){}
    }

    function nextIndex(rowKey){
      return state[rowKey].slotsFilled.findIndex(v => v === false);
    }
    function isNextSlot(rowKey, idx){ return nextIndex(rowKey) === idx; }

    function allRowsDoneAndSpoken(){
      return ROWS.every(r => state[r.key].complete && state[r.key].slovoPlayed);
    }

    function clearHighlights(){
      document.querySelectorAll(".slot").forEach(s=>s.classList.remove("dropOk","dropBad"));
    }

    function build(){
      const grid = document.getElementById("gameGrid");
      grid.innerHTML = "";

      ROWS.forEach(row => {
        state[row.key] = {
          placedCount: 0,
          slotsFilled: Array(row.word.length).fill(false),
          slovoPlayed: false,
          complete: false
        };

        // LEFT (letters) — один рядок гри = одна ліва клітинка
        const left = document.createElement("div");
        left.className = "cell left";

        const lettersWrap = document.createElement("div");
        lettersWrap.className = "letters";
        lettersWrap.dataset.row = row.key;

        const tokens = shuffle(row.letterSounds.map((ls, idx) => ({
          tokenId: `${row.key}_t${idx}`,
          ch: ls.ch,
          audio: ls.audio,
          rowKey: row.key
        })));

        tokens.forEach(tok => {
          const el = document.createElement("div");
          el.className = "letter";
          el.textContent = tok.ch;
          el.setAttribute("draggable","true");
          el.dataset.tokenId = tok.tokenId;
          el.dataset.rowKey = tok.rowKey;
          el.dataset.ch = tok.ch;
          el.dataset.audio = tok.audio;

          // клік = звук букви
          el.addEventListener("click", () => {
            if(el.classList.contains("locked")) return;
            playAudioById(tok.audio);
          });

          el.addEventListener("dragstart", (e) => {
            if(el.classList.contains("locked")) { e.preventDefault(); return; }
            dragTokenId = tok.tokenId;
            el.classList.add("dragging");
            playAudioById(tok.audio);
            e.dataTransfer.setData("text/plain", tok.tokenId);
            e.dataTransfer.effectAllowed = "move";
          });

          el.addEventListener("dragend", () => {
            el.classList.remove("dragging");
            dragTokenId = null;
            clearHighlights();
          });

          lettersWrap.appendChild(el);
        });

        left.appendChild(lettersWrap);

        // RIGHT (image + targets + belowBar)
        const right = document.createElement("div");
        right.className = "cell right";

        const img = document.createElement("img");
        img.className = "pic";
        img.alt = "Музичний інструмент";
        img.src = row.imageSrc;

        const targetsWrap = document.createElement("div");
        targetsWrap.className = "targetsWrap";

        const targets = document.createElement("div");
        targets.className = "targets";

        row.word.forEach((expectedCh, i) => {
          const slot = document.createElement("div");
          slot.className = "slot";
          slot.dataset.row = row.key;
          slot.dataset.index = String(i);
          slot.dataset.expected = expectedCh;

          slot.addEventListener("dragover", (e) => {
            const st = state[row.key];
            if(st.complete) return;
            if(!dragTokenId) return;

            e.preventDefault();

            const tokenEl = document.querySelector(`.letter[data-token-id="${dragTokenId}"]`);
            if(!tokenEl) return;

            const tokenRow = tokenEl.dataset.rowKey;
            const tokenCh  = tokenEl.dataset.ch;

            const ok = (tokenRow === row.key) && isNextSlot(row.key, i) && (tokenCh === expectedCh) && !slot.classList.contains("filled");

            slot.classList.toggle("dropOk", ok);
            slot.classList.toggle("dropBad", !ok);
          });

          slot.addEventListener("dragleave", () => {
            slot.classList.remove("dropOk","dropBad");
          });

          slot.addEventListener("drop", (e) => {
            e.preventDefault();
            slot.classList.remove("dropOk","dropBad");

            const tokenId = e.dataTransfer.getData("text/plain") || dragTokenId;
            const tokenEl = document.querySelector(`.letter[data-token-id="${tokenId}"]`);
            if(!tokenEl) return;

            const st = state[row.key];
            if(st.complete) return;

            const tokenRow = tokenEl.dataset.rowKey;
            const tokenCh  = tokenEl.dataset.ch;

            const ok = (tokenRow === row.key) && isNextSlot(row.key, i) && (tokenCh === expectedCh) && !slot.classList.contains("filled");

            if(!ok){
              tokenEl.classList.add("shake");
              setTimeout(()=>tokenEl.classList.remove("shake"), 260);
              return;
            }

            slot.classList.add("filled");
            slot.appendChild(tokenEl);
            tokenEl.classList.add("locked");
            tokenEl.setAttribute("draggable","false");

            st.slotsFilled[i] = true;
            st.placedCount++;

            if(st.placedCount === row.word.length){
              st.complete = true;
              showBelowBar(row.key);
            }
          });

          targets.appendChild(slot);
        });

        // belowBar під кружечками
        const belowBar = document.createElement("div");
        belowBar.className = "belowBar";
        belowBar.id = `below_${row.key}`;

        targetsWrap.appendChild(targets);
        targetsWrap.appendChild(belowBar);

        right.appendChild(img);
        right.appendChild(targetsWrap);

        grid.appendChild(left);
        grid.appendChild(right);
      });
    }

    function showBelowBar(rowKey){
      const row = ROWS.find(r => r.key === rowKey);
      const st  = state[rowKey];
      const below = document.getElementById(`below_${rowKey}`);
      below.innerHTML = "";

      const soundBtn = document.createElement("button");
      soundBtn.className = "soundIcon";
      soundBtn.type = "button";
      soundBtn.title = "Прослухати слово";
      soundBtn.textContent = "🔊"; // ✅ лише значок

      const check = document.createElementNS("http://www.w3.org/2000/svg", "svg");
      check.setAttribute("viewBox", "0 0 24 24");
      check.classList.add("check");
      check.innerHTML = `<path fill="${getComputedStyle(document.documentElement).getPropertyValue('--good') || '#1fa84a'}"
        d="M9.0 16.2l-3.5-3.5L4 14.2l5 5L20 8.2 18.6 6.8z"/>`;

      soundBtn.addEventListener("click", () => {
        playAudioById(row.slovoAudio);
        st.slovoPlayed = true;
        check.classList.add("show");
        soundBtn.disabled = true;

        if(allRowsDoneAndSpoken()){
          document.getElementById("finalBar").classList.add("show");
        }
      });

      below.appendChild(soundBtn);
      below.appendChild(check);
    }

    function resetGame(){
      document.getElementById("finalBar").classList.remove("show");
      dragTokenId = null;
      build();
    }

    document.getElementById("zavdahja").addEventListener("click", () => playAudioById("taskAudio"));
    document.getElementById("resetTop").addEventListener("click", resetGame);
    document.getElementById("resetBottom").addEventListener("click", resetGame);
    document.getElementById("winBtn").addEventListener("click", () => alert("Це — успіх!"));

    document.getElementById("taskText").textContent = TASK_TEXT;
    build();
  </script>
