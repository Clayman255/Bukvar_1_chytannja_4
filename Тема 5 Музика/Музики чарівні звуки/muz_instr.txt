  const player = document.getElementById("player");

  function normalizeSrc(src){
    try { return new URL(src, window.location.href).href; }
    catch { return String(src || ""); }
  }

  function stopAudio(){
    player.pause();
    try { player.currentTime = 0; } catch {}
  }

  function toggleAudio(src){
    if(!src) return;
    const target = normalizeSrc(src);
    const current = normalizeSrc(player.currentSrc || player.src || "");
    const isSame = current && (current === target);
    const isPlaying = !player.paused && !player.ended;

    if(isSame && isPlaying){
      stopAudio();
      return;
    }

    player.pause();
    try { player.currentTime = 0; } catch {}
    player.src = src;
    player.play().catch(()=>{});
  }

  function imgToZvu(imgPath){

  function applyTransforms(items){
    items.forEach(el => {
      const s = el.__pos;
      el.style.transform = `translate(calc(-50% + ${s.dx}px), calc(-50% + ${s.dy}px))`;

  // ✅ ПОВНИЙ bbox для елемента: інструмент + мовленнєве поле + кнопка 🎵
  // (бо absolute-діти НЕ збільшують bbox батька, тому раніше solver міг "не бачити" накладання)
  function getFullBox(itemEl){
    const inst = itemEl.querySelector(".inst");
    const bubble = itemEl.querySelector(".bubble");
    const note = itemEl.querySelector(".noteBtn");
    const parts = [inst, bubble, note].filter(Boolean);

    // fallback
    if(!parts.length) return itemEl.getBoundingClientRect();

    const rects = parts.map(p => p.getBoundingClientRect());
    const left   = Math.min(...rects.map(r => r.left));
    const top    = Math.min(...rects.map(r => r.top));
    const right  = Math.max(...rects.map(r => r.right));
    const bottom = Math.max(...rects.map(r => r.bottom));
    return { left, top, right, bottom, width: right-left, height: bottom-top };
  }

  function solveLayoutNoOverlap({
    iterations = 220,
    gap = MIN_GAP_PX,
    boundsMargin = 10
  } = {}){
    const items = Array.from(circle.querySelectorAll(".item"));
    if(items.length < 2) return;

    const box = circle.getBoundingClientRect();
    const minX = box.left + boundsMargin;
    const maxX = box.right - boundsMargin;
    const minY = box.top + boundsMargin;
    const maxY = box.bottom - boundsMargin;

    const dyrRect = document.getElementById("dyr").getBoundingClientRect();

    applyTransforms(items);

    for(let it=0; it<iterations; it++){
      let moved = false;

      // актуальні bbox-и (повні)
      const rects = items.map(getFullBox);
      const centers = rects.map(r => ({ x:(r.left+r.right)/2, y:(r.top+r.bottom)/2 }));

      // 1) розсування між елементами з урахуванням gap
      for(let i=0;i<items.length;i++){
        for(let j=i+1;j<items.length;j++){
          const a = rects[i], b = rects[j];
          const overlapX = Math.min(a.right, b.right) - Math.max(a.left, b.left);
          const overlapY = Math.min(a.bottom, b.bottom) - Math.max(a.top, b.top);

          // якщо перетинаються або ближче ніж gap по обох осях
          if(overlapX > -gap && overlapY > -gap){
            // наскільки "втиснуті" з урахуванням необхідного зазору
            const penX = overlapX + gap;
            const penY = overlapY + gap;

            const ci = centers[i], cj = centers[j];
            const signX = (cj.x >= ci.x) ? 1 : -1;
            const signY = (cj.y >= ci.y) ? 1 : -1;

            // штовхаємо по осі з МЕНШИМ проникненням (щоб мінімально рухати)
            if(penX < penY){
              const push = penX / 2;
              items[i].__pos.dx -= signX * push;
              items[j].__pos.dx += signX * push;
            } else {
              const push = penY / 2;
              items[i].__pos.dy -= signY * push;
              items[j].__pos.dy += signY * push;
            }
            moved = true;
          }
        }
      }

      // 2) не наїжджати на диригента (як на перешкоду з gap)
      applyTransforms(items);
      const rects2 = items.map(getFullBox);
      for(let i=0;i<items.length;i++){
        const r = rects2[i];

        const overlapX = Math.min(r.right, dyrRect.right) - Math.max(r.left, dyrRect.left);
        const overlapY = Math.min(r.bottom, dyrRect.bottom) - Math.max(r.top, dyrRect.top);

        if(overlapX > -gap && overlapY > -gap){
          const penX = overlapX + gap;
          const penY = overlapY + gap;

          const cx = (r.left+r.right)/2;
          const dxSign = (cx >= (dyrRect.left+dyrRect.right)/2) ? 1 : -1;
          const cy = (r.top+r.bottom)/2;
          const dySign = (cy >= (dyrRect.top+dyrRect.bottom)/2) ? 1 : -1;

          if(penX < penY){
            items[i].__pos.dx += dxSign * penX;
          } else {
            items[i].__pos.dy += dySign * penY;
          }
          moved = true;
        }
      }

      // 3) утримати в межах білого поля (не обрізати — просто посунути)
      applyTransforms(items);
      const rects3 = items.map(getFullBox);
      for(let i=0;i<items.length;i++){
        const r = rects3[i];
        let shiftX = 0, shiftY = 0;

        if(r.left < minX) shiftX = (minX - r.left);
        if(r.right > maxX) shiftX = (maxX - r.right);

        if(r.top < minY) shiftY = (minY - r.top);
        if(r.bottom > maxY) shiftY = (maxY - r.bottom);

        if(shiftX || shiftY){
          items[i].__pos.dx += shiftX;
          items[i].__pos.dy += shiftY;
          moved = true;
        }
      }

      applyTransforms(items);

      // якщо нічого не рухали — все ок
      if(!moved) break;
    }

    // фінальна перевірка/піджим до меж
    applyTransforms(items);
    const rectsF = items.map(getFullBox);
    rectsF.forEach((r, i) => {
      let shiftX = 0, shiftY = 0;
      if(r.left < minX) shiftX = (minX - r.left);
      if(r.right > maxX) shiftX = (maxX - r.right);
      if(r.top < minY) shiftY = (minY - r.top);
      if(r.bottom > maxY) shiftY = (maxY - r.bottom);
      if(shiftX || shiftY){
        items[i].__pos.dx += shiftX;
        items[i].__pos.dy += shiftY;
      }
    });
    applyTransforms(items);
  }

  function buildCircle(){
    circle.querySelectorAll(".item").forEach(n => n.remove());

    const n = instruments.length;
    const rect = circle.getBoundingClientRect();

    let R = Math.max(150, Math.min(235, rect.width * 0.255));

    const startDeg = -90;
    const stepDeg = 360 / n;

    const refScale = instruments.find(x => x.id === 8).base * (tweak[8] || 1) * GLOBAL_SCALE;
    setConductorScale(refScale);

    const clamp = (v) => Math.max(0.30, Math.min(2.10, v));
    function rawScale(ins){
      let s = ins.base * (tweak[ins.id] || 1);
      if(ins.id === 1){ s = refScale; }
      if(ins.id === 5){ s = s * 1.10; }
      s *= GLOBAL_SCALE;
      return clamp(s);
    }

    const accordionScale = rawScale(instruments.find(x => x.id === 1));
    const groupToAccordion = new Set([2,4,6,7]);
    const drimbaShrink = 4/9;

    let domraScale = clamp(accordionScale);

    instruments.forEach((ins, i) => {
      const deg = startDeg + stepDeg * i;
      const rad = deg * Math.PI / 180;

      const x = Math.cos(rad) * R;
      const y = Math.sin(rad) * R;

      const item = document.createElement("div");
      item.className = "item";
      item.__pos = {dx:x, dy:y};
      item.style.transform = `translate(calc(-50% + ${x}px), calc(-50% + ${y}px))`;

      const inst = document.createElement("div");
      inst.className = "inst";

      let s = rawScale(ins);
      if(groupToAccordion.has(ins.id)) s = accordionScale;
      if(ins.id === 3) s = accordionScale * drimbaShrink;

      s = clamp(s);
      inst.style.transform = `scale(${s})`;

      const bubbleSound = imgToZvu(ins.img);

      inst.innerHTML = `
        <div class="bubble" id="bubble_${ins.id}" data-audio="${bubbleSound}">${ins.bubble}</div>
        <img src="${ins.img}" alt="${ins.name}">
        <button class="noteBtn" id="note_${ins.id}" type="button" title="Прослухати звук інструмента" data-audio="${ins.sound}">
          <span>🎵</span>
        </button>
      `;

      const bubble = inst.querySelector(`#bubble_${ins.id}`);
      bubble.addEventListener("click", (e) => {
        e.stopPropagation();
        toggleAudio(bubble.dataset.audio);
      });

      if(ins.id === 1 || ins.id === 5){
        const bubbleScale = refScale / s;
        bubble.style.transform = `translate(-50%, -18px) scale(${bubbleScale})`;
      }
      if(ins.id === 2 || ins.id === 6){
        const bubbleScale = domraScale / s;
        bubble.style.transform = `translate(-50%, -18px) scale(${bubbleScale})`;
      }

      const noteBtn = inst.querySelector(`#note_${ins.id}`);
      if(ins.id === 5){
        const btnScale = refScale / s;
        noteBtn.style.setProperty("--btnScale", btnScale.toFixed(4));
      } else {
        noteBtn.style.setProperty("--btnScale", "1");
      }

      inst.addEventListener("click", () => openModal(ins));
      noteBtn.addEventListener("click", (e) => {
        e.stopPropagation();
        toggleAudio(noteBtn.dataset.audio);
      });

      item.appendChild(inst);
      circle.appendChild(item);
    });

    // ✅ головне: гарантовано без накладань і з відстанню
    requestAnimationFrame(() => {
      solveLayoutNoOverlap({iterations: 260, gap: MIN_GAP_PX, boundsMargin: 10});
      requestAnimationFrame(() => {
        // ще один контрольний прохід (після того як браузер усе доміряє)
        solveLayoutNoOverlap({iterations: 180, gap: MIN_GAP_PX, boundsMargin: 10});
      });
    });
  }
