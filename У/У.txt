<script>
  const itemsData = [
    { id: 1, img: 'Img_1.png', correct: true },
    { id: 2, img: 'Img_2.png', correct: true },
    { id: 3, img: 'Img_3.png', correct: false },
    { id: 4, img: 'Img_4.png', correct: false },
    { id: 5, img: 'Img_5.png', correct: true },
    { id: 6, img: 'Img_6.png', correct: true },
    { id: 7, img: 'Img_7.png', correct: true },
    { id: 8, img: 'Img_8.png', correct: true }
  ];

  const board = document.getElementById('board');
  const basketContent = document.getElementById('basketContent');
  const dropzone = document.getElementById('dropzone');
  const audioTask = document.getElementById('audioTask');
  const winMsg = document.getElementById('winMsg');

  let correctCount = 0;
  const totalCorrect = itemsData.filter(i => i.correct).length;

  function initGame() {
    board.innerHTML = '';
    basketContent.innerHTML = '';
    winMsg.style.display = 'none';
    correctCount = 0;

    [...itemsData].sort(() => 0.5 - Math.random()).forEach(item => {
      const el = document.createElement('div');
      el.className = 'item';
      el.draggable = true;
      el.dataset.id = item.id;
      el.dataset.correct = item.correct;
      el.innerHTML = `<img src="${item.img}">`;

      el.addEventListener('dragstart', e => {
        e.dataTransfer.setData('text/plain', item.id);
        setTimeout(() => el.style.opacity = '0.5', 0);
      });

      el.addEventListener('dragend', () => el.style.opacity = '1');
      board.appendChild(el);
    });
  }

  dropzone.addEventListener('dragover', e => e.preventDefault());

  dropzone.addEventListener('drop', e => {
    e.preventDefault();
    const id = e.dataTransfer.getData('text/plain');
    const dragged = document.querySelector(`.item[data-id="${id}"]`);
    if (!dragged) return;

    if (dragged.dataset.correct === 'true') {
      basketContent.appendChild(dragged.cloneNode(true));
      dragged.remove();
      correctCount++;
      if (correctCount === totalCorrect) winMsg.style.display = 'block';
    } else {
      dragged.classList.add('error');
      setTimeout(() => dragged.classList.remove('error'), 400);
    }
  });

  document.getElementById('btnSound').onclick = () => {
    audioTask.currentTime = 0;
    audioTask.play();
  };

  document.getElementById('btnReset').onclick = initGame;

  initGame();
</script>
