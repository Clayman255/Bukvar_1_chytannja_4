  <script>
    // Масив рівнів (порядок букв фіксований)
    const levels = [
      { id: 1, correct: "САД",      scrambled: ["А","С","Д"] },
      { id: 2, correct: "ВОДА",     scrambled: ["В","О","А","Д"] },
      { id: 3, correct: "ПРЕДМЕТ",  scrambled: ["П","Р","Е","Д","Е","М","Т"] },
      { id: 4, correct: "ЛІД",      scrambled: ["Л","Д","І"] },
      { id: 5, correct: "ДОМІНО",   scrambled: ["О","Д","М","І","Н","О"] },
      { id: 6, correct: "КРОКОДИЛ", scrambled: ["К","Р","О","К","О","Л","И","Д"] },
      { id: 7, correct: "МЕД",      scrambled: ["Д","Е","М"] },
      { id: 8, correct: "СОДА",     scrambled: ["С","О","А","Д"] },
      { id: 9, correct: "ДИКТОР",   scrambled: ["И","Д","К","Т","О","Р"] }
    ];

    const gameBoard = document.getElementById("game-board");
    const afterWin = document.getElementById("afterWin");
    const btnAgain = document.getElementById("btnAgain");
    const btnSound = document.getElementById("btnSound");

    // Отримуємо елемент аудіо за id="au"
    const au = document.getElementById("au");
    const ok = document.getElementById("ok");
    const no = document.getElementById("no");

    let draggedItem = null;

    function stopAll(){
      document.querySelectorAll("audio").forEach(a => { try{ a.pause(); a.currentTime=0; }catch(e){} });
    }
    
    function playAudio(a){
      if(!a) return;
      stopAll();
      a.currentTime = 0;
      a.play().catch(()=>{});
    }

    // При натисканні на кнопку запускаємо змінну au
    btnSound.addEventListener("click", ()=> playAudio(au));

    function initGame(){
      afterWin.style.display = "none";
      gameBoard.innerHTML = "";

      levels.forEach((level, index) => {
        const row = document.createElement("div");
        row.className = "task-row";
        row.id = `row-${index}`;
        row.dataset.rowIndex = index;

        // Ліва частина: слоти
        const dropZoneArea = document.createElement("div");
        dropZoneArea.className = "drop-zone-area";

        for(let i=0;i<level.correct.length;i++){
          const slot = document.createElement("div");
          slot.className = "slot";
          slot.dataset.expected = level.correct[i];
          slot.dataset.rowIndex = index; 
          
          slot.addEventListener("dragover", handleDragOver);
          slot.addEventListener("drop", handleDrop);
          slot.addEventListener("dragenter", handleDragEnter);
          slot.addEventListener("dragleave", handleDragLeave);

          dropZoneArea.appendChild(slot);
        }

        // Права частина: букви
        const lettersArea = document.createElement("div");
        lettersArea.className = "letters-area";
        lettersArea.id = `origin-${index}`;

        lettersArea.addEventListener("dragover", handleDragOver);
        lettersArea.addEventListener("drop", handleReturnDrop);

        // Букви додаємо без перемішування
        level.scrambled.forEach((ch, charIndex)=>{
          const letter = document.createElement("div");
          letter.className = "letter-box";
          letter.draggable = true;
          letter.textContent = ch;
          letter.id = `l-${index}-${charIndex}`;
          letter.dataset.origin = `origin-${index}`;
          letter.dataset.rowIndex = index; 
          
          letter.addEventListener("dragstart", handleDragStart);
          lettersArea.appendChild(letter);
        });

        row.appendChild(dropZoneArea);
        row.appendChild(lettersArea);
        gameBoard.appendChild(row);
      });
    }

    function handleDragStart(e){
      draggedItem = this;
      e.dataTransfer.effectAllowed = "move";
      e.dataTransfer.setData("text/plain", this.id);
      setTimeout(()=> this.style.opacity = "0.5", 0);
    }

    function handleDragOver(e){
      e.preventDefault();
      e.dataTransfer.dropEffect = "move";
    }

    function handleDragEnter(e){
      e.preventDefault();
      if(this.classList.contains("slot") && !this.hasChildNodes() && !this.classList.contains("locked")){
        this.classList.add("over");
      }
    }

    function handleDragLeave(e){
      if(this.classList.contains("slot")){
        this.classList.remove("over");
      }
    }

    function handleDrop(e){
      e.preventDefault();
      this.classList.remove("over");

      if(!draggedItem) return;
      if(this.classList.contains("locked")) { resetDrag(); return; }
      if(!this.classList.contains("slot")) { resetDrag(); return; }
      if(this.hasChildNodes()) { resetDrag(); return; }

      // Перевірка: чи належить буква цьому рядку?
      const itemRow = draggedItem.dataset.rowIndex;
      const slotRow = this.dataset.rowIndex;

      if(itemRow !== slotRow){
        playAudio(no);
        this.classList.add("fail-anim");
        setTimeout(()=> this.classList.remove("fail-anim"), 260);
        returnToOrigin(draggedItem);
        return;
      }

      // Перевірка: чи правильна буква
      const expected = this.dataset.expected;
      const got = draggedItem.textContent;

      if(got !== expected){
        playAudio(no);
        this.classList.add("fail-anim");
        setTimeout(()=> this.classList.remove("fail-anim"), 260);
        returnToOrigin(draggedItem);
        return;
      }

      this.appendChild(draggedItem);
      draggedItem.style.opacity = "1";
      this.classList.add("locked");

      playAudio(ok);
      draggedItem.classList.add("success-anim");
      setTimeout(()=> draggedItem.classList.remove("success-anim"), 320);

      draggedItem.draggable = false;
      draggedItem.classList.add("used");
      draggedItem = null;

      const row = this.closest(".task-row");
      checkRow(row);
      checkAllWin(); 
    }

    function handleReturnDrop(e){
      e.preventDefault();
      if(!draggedItem) return;
      if(draggedItem.classList.contains("used")){ resetDrag(); return; }
      returnToOrigin(draggedItem);
    }

    function returnToOrigin(item){
       const originId = item.dataset.origin;
       const origin = document.getElementById(originId);
       if(origin) origin.appendChild(item);
       resetDrag();
    }

    function resetDrag(){
      if(draggedItem){ draggedItem.style.opacity = "1"; }
      draggedItem = null;
    }

    function checkRow(row){
      const slots = row.querySelectorAll(".slot");
      let current = "";
      let full = true;
      slots.forEach(s=>{
        if(s.children.length){ current += s.children[0].textContent; }
        else{ full = false; }
      });
      const idx = parseInt(row.id.split("-")[1], 10);
      const correct = levels[idx].correct;
      if(full && current === correct){
        row.classList.add("correct-row");
      }
    }

    function checkAllWin(){
      const allSlots = document.querySelectorAll(".slot");
      const locked = document.querySelectorAll(".slot.locked").length;

      if(allSlots.length > 0 && locked === allSlots.length){
        afterWin.style.display = "flex";
        playAudio(ok);
        window.scrollTo({ top: 0, behavior: 'smooth' });
      }
    }

    btnAgain.addEventListener("click", initGame);
    document.addEventListener("dragend", ()=> resetDrag());

    initGame();
  </script>
