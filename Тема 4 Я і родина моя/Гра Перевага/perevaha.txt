<script>
  const images = Array.from({ length: 6 }, (_, i) => `img${i+1}.png`);

  const game = document.getElementById("game");
  const endBlock = document.getElementById("end");

  let lastOpenedCard = null;
  let order = [];
  let step = 0;

  function shuffle(arr){
    for (let i = arr.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
    return arr;
  }

  function createCard(){
    const card = document.createElement("div");
    card.className = "card";

    const inner = document.createElement("div");
    inner.className = "card-inner";

    inner.innerHTML = `
      <div class="face front"></div>
      <div class="face back"></div>
    `;

    card.appendChild(inner);
    card.addEventListener("click", () => openNext(card));
    return card;
  }

  function initGame(){
    game.innerHTML = "";
    lastOpenedCard = null;
    order = shuffle([...Array(6).keys()]);
    step = 0;
    endBlock.style.display = "none";

    for(let i=0;i<6;i++){
      game.appendChild(createCard());
    }
  }

  function openNext(card){
    if(step >= order.length) return;

    if(lastOpenedCard){
      lastOpenedCard.classList.add("hidden");
    }

    const back = card.querySelector(".back");
    back.innerHTML = `<img src="${images[order[step]]}" alt="">`;

    card.classList.add("flipped");
    lastOpenedCard = card;
    step++;

    if(step === order.length){
      setTimeout(() => {
        endBlock.style.display = "block";
      }, 650);
    }
  }

  function restart(){
    initGame();
  }

  document.getElementById("sound").onclick = () => {
    const a = document.getElementById("audio");
    a.currentTime = 0;
    a.play();
  };

  initGame();
</script>
